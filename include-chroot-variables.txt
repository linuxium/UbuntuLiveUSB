# Linuxium's scripts to create a custom Ubuntu ISO or multiboot Ubuntu LiveUSB

CANONICAL_ISO="ubuntu-16.10-desktop-amd64.iso"
PATH_TO="/tmp"

wget https://www.kernel.org/finger_banner -O ${PATH_TO}/finger_banner
VERSION=$(cat ${PATH_TO}/finger_banner | sed -n '2p' | awk -F':' '{print $2}' | sed 's/ //g')

LINUXIUM_KERNEL_RELEASE=${VERSION}
LINUXIUM_KERNEL_VERSION="stable"

LINUXIUM_ISO="ubuntu-16.10-${LINUXIUM_KERNEL_VERSION}.iso"
LINUXIUM_VOLID="Ubuntu 16.10 Custom Audio"
